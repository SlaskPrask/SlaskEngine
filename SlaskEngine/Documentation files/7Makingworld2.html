Functions and methods provided by the engine are in bold text.


world2.cpp


#include "world2.h"


World2::World2()

{

   spriteSet(0)->load();

   

   setSize(2000,800);

   player=at(new Player(),width/2,height/2);

   

   cam.setSize(600,400);

   cam.setLimits(width,height);

   cam.setBounds(120,100);

   cam.follow(player);

   cam.setPositionCentered(player);

   cam.activate();

}


World2::~World2()

{

   spriteSet(0)->unload();

}
